    public void changeUI()
    {
    pane = getContentPane();
    pane.setLayout(new BorderLayout());
    boardPanel = new JPanel();
    boardPanel.repaint();
    GridLayout layout = new GridLayout( 3, 3, 0, 0 );
    boardPanel.setLayout( layout );

    board = new Square[ 3 ][ 3 ];

    // When creating a Square, the location argument to the
    // constructor is a value from 0 to 8 indicating the
    // position of the Square on the board. Values 0, 1,
    // and 2 are the first row, values 3, 4, and 5 are the
    // second row. Values 6, 7, and 8 are the third row.
    for ( int row = 0; row < board.length; row++ )
    {
    for ( int col = 0;
    col < board[ row ].length; col++ ) {
    board[ row ][ col ] =
    new Square( ' ', row * 3 + col );
    board[ row ][ col ].addMouseListener(
    new SquareListener(this,
    board[ row ][ col ] ) );
    boardPanel.add( board[ row ][ col ] );
    board[ row ][ col ].repaint();
    }
    }

    id = new JTextField();
    id.setEditable( false );

    pane.add( id, BorderLayout.NORTH );
    mainPanel = new MainPanel();

    pane.add(mainPanel, BorderLayout.CENTER);

    display = new JTextArea( 4, 30 );
    display.setEditable( false );

    pane.add( new JScrollPane(display),BorderLayout.SOUTH );

    if(setup.success){
    IPAddress=setup.IPText.getText();
    portNum=setup.portText.getText();
    System.out.println(IPAddress);
    System.out.println(portNum);
    start();

    }

    // Make connection to server and get associated streams.
    // Start separate thread to allow this applet to
    // continually update its output in text area display.
    public void start()
    {
    if(setup.success){

    try {
    connection = new Socket(
    InetAddress.getByName(IPAddress), Integer.parseInt(portNum));

    input = new DataInputStream(
    connection.getInputStream() );
    output = new DataOutputStream(
    connection.getOutputStream() );
    }
    catch(NullPointerException e0){}

    catch(NumberFormatException e1){}

    catch(ConnectException e2){
    JOptionPane.showMessageDialog(null,"Connection Refused.. Server maybe down...","Cannot locate Server", JOptionPane.ERROR_MESSAGE);
    }

    catch (IOException e ) {
    JOptionPane.showMessageDialog(null,"Cannot connect to server", "Server Error", JOptionPane.ERROR_MESSAGE);
    }

    outputThread.start();
    }//if
    }

    // Control thread that allows continuous update of the
    // text area display.
    public void run()
    {

    System.out.println("Running in run method");

    // First get player's mark (X or O)
    try {
    String str="";
    myMark = input.readChar();
    System.out.println("myMark: " +myMark);
    id.setText( "You are player \"" + myMark + "\"" );
    myTurn = ( myMark == 'X' ? true : false );
    }
    catch ( IOException e ) {
    e.printStackTrace();
    }

    // Receive messages sent to client
    while (true) {
    System.out.println("In while loop");
    try {
    String s = input.readUTF();
    System.out.println("Msg: "+s);
    processMessage(s);
    System.out.println("Ended");
    }
    catch ( IOException e ) {
    e.printStackTrace();
    }
    }//while

    } //run

    // Process messages sent to client
    public void processMessage( String s )
    {
    if ( s.equals( "Valid move." ) ) {
    display.append( "Valid move, please wait.\n" );
    currentSquare.setMark( myMark );
    currentSquare.repaint();

    }
    else if ( s.equals( "Invalid move, try again" ) ) {
    display.append( s + "\n" );
    myTurn = true;
    }
    else if ( s.equals("Opponent moved") ) {
    try {
    int loc = input.readInt();

    board[ loc / 3 ][ loc % 3 ].setMark(
    (myMark == 'X' ? 'O' : 'X' ));
    board[ loc / 3 ][ loc % 3 ].repaint();

    display.append("Opponent moved. Your turn.\n" );

    myTurn = true;

    }
    catch ( IOException e ) {
    e.printStackTrace();
    }

    }
    else if(s.equals("Server: Game Ended")){

    mainPanel.newGameBtn.setEnabled(true);
    // mainPanel.endGameBtn.setEnabled(false);
    myTurn=false;
    }
    else if(s.equals("Server: New Game")){
    System.out.println("start a new game");
    prepareNewGame();

    }
    else if(s.equals("Server: Grant Turn O")){

    if(myMark=='O')
    myTurn=true;
    else
    myTurn=false;
    display.append("O to move now. X please wait...\n");

    }
    else if(s.equals("Server: Grant Turn X") || s.equals("Grant Turn X")){

    if(myMark=='X'){
    myTurn=true;
    }
    else{
    myTurn=false;
    }/// end if else

    if(s.equals("Grant Turn X")){
    display.append("X to move now...\n");
    }
    else{
    display.append("X to move now. O please wait...\n");

    }//end if else

    }
    else if(s.equals("Server moved")){
    int loc=-1;
    try{
    loc = input.readInt();
    }
    catch(IOException e){}

    System.out.println("loc "+loc);
    board[ loc / 3 ][ loc % 3 ].setMark('O');
    board[ loc / 3 ][ loc % 3 ].repaint();
    myTurn=true;
    display.append("Server moved. Your turn.\n");

    }
    else if(s.equals("Current player moved")){

    int loc=-1;
    try{
    loc = input.readInt();
    }
    catch(IOException e){}

    board[ loc / 3 ][ loc % 3 ].setMark('X');
    board[ loc / 3 ][ loc % 3 ].repaint();
    myTurn=false;

    }

    else
    display.append( s + "\n" );

    display.setCaretPosition(display.getText().length());
    }

    {
    super.paintComponent( g );
    g.setFont(new Font("Arial", Font.BOLD, 14));

    g.drawRect( 0, 0, 29, 29 );
    g.drawString(String.valueOf(mark), 11, 20);

    public void setSquare(char m, int loc){
    mark = m;
    location = loc;
    }

    import java.awt.*;

    public class TicTacToeServer extends JFrame {

    private char board[];
    private boolean xMove;
    private JTextArea output;
    private Player players[];
    private ServerSocket server;
    private int currentPlayer;

    final int SERVER = 3;

    boolean playWithServer=false;
    boolean serverTurn=false;
    boolean draw=false;
    boolean gameOver=false;
    char winnerId='E';
    int numPlayers=-1;

    public TicTacToeServer()
    {
    super( "Tic-Tac-Toe Server" );

    board = new char[9];
    xMove = true;

    players = new Player[2];
    currentPlayer = 0;

    numPlayers=players.length;

    //init server board
    for(int i=0; i<9; i++){
    board[i]= ' ';
    }

    // set up ServerSocket
    try {
    server = new ServerSocket(5000, 2 );
    }
    catch( IOException e ) {
    e.printStackTrace();
    System.exit( 1 );
    }

    output = new JTextArea();
    getContentPane().add( output, BorderLayout.CENTER );
    output.setEditable(false);
    output.setText( "Server awaiting connections\n" );

    setSize( 300, 300 );
    show();
    }

    // wait for two connections so game can be played
    public void execute()
    {

    try{

    for ( int i = 0; i < numPlayers ; i++ ) {
    players[i] = new Player(server.accept(),this,i);
    players[ i ].start();
    }

    }
    catch( IOException e ) {
    e.printStackTrace();
    System.exit( 1 );
    }

    // Player X is suspended until Player O connects.
    // Resume player X now.
    synchronized ( players[ 0 ] ) {
    players[ 0 ].threadSuspended = false;
    players[ 0 ].notify();
    }

    }

    public void display( String s )
    {
    output.append( s + "\n" );
    }

    // Determine if a move is valid.
    // This method is synchronized because only one move can be
    // made at a time.
    public synchronized boolean validMove( int loc, int player)
    {
    boolean status=false;

    while ( player != currentPlayer ) {
    try {
    wait();
    }
    catch( InterruptedException e ) {
    e.printStackTrace();
    }
    }

    if ( !isOccupied( loc ) ) {

    board[loc] =(currentPlayer == 0 ? 'X' : 'O' );
    if(playWithServer){
    players[currentPlayer].currentPlayerMoved(loc);
    }
    else{
    currentPlayer = ( currentPlayer + 1 ) % 2;
    players[currentPlayer].otherPlayerMoved( loc );
    notify(); // tell waiting player to continue
    }

    status=true;

    }
    else{
    status=false;
    }

    return status;
    }

    public boolean isOccupied( int loc )
    {
    if ( board[ loc ] == 'X' || board [ loc ] == 'O' ){
    return true;
    }
    else{
    return false;
    }
    }

    //This method decides whether the game is a win, draw or lose
    public boolean gameOver(char ch)
    { boolean status=false;
    char player='E';
    int count=0;

    player=checkWinner(board, ch);

    // a win is detected
    if(player==ch){
    status=true;
    draw=false;
    }
    else{ //check whether it is a draw

    for(int i=0; i<9; i++){
    if(isOccupied(i)){
    count++;
    }
    }

    if(count==9){
    status=true;
    draw=true;
    }
    }

    return status;
    }

    //This methods checks whether a player has win
    private char checkWinner(char board[], char ch)
    {

    char winner='E';

    //horizontal check
    if((board[0]==ch && board[1]==ch) && board[2]==ch){
    winner=ch;
    }
    else if((board[3]==ch && board[4]==ch) && board[5]==ch){
    winner=ch;
    }
    else if((board[6]==ch && board[7]==ch) && board[8]==ch){
    winner=ch;
    }

    //vertical check
    else if((board[0]==ch && board[3]==ch) && board[6]==ch){
    winner=ch;
    }
    else if((board[1]==ch && board[4]==ch) && board[7]==ch){
    winner=ch;
    }
    else if((board[2]==ch && board[5]==ch) && board[8]==ch){
    winner=ch;
    }

    //Diagonal check, \ direction
    else if((board[0]==ch && board[4]==ch) && board[8]==ch){
    winner=ch;
    }

    // Diagonal check, / direction
    else if((board[2]==ch && board[4]==ch) && board[6]==ch){
    winner=ch;
    }

    winnerId=winner; //use in startNewGame later on

    return winner;

    }

    //This method send the results to the players indicating who win/lose or draw
    public void sendResults(char ch){

    char winner='E';
    char loser='E';

    if(ch=='X'){ // X wins
    winner='X';
    loser='O';
    }
    else if(ch=='O'){ // O wins
    winner='O';
    loser='X';
    }
    else if(ch=='D'){ //a draw

    try{
    for (int i=0; i < numPlayers; i++){
    players[i].output.writeUTF("The game is a draw");
    }
    }
    catch(IOException e){
    e.printStackTrace();
    }
    }

    //write the win/lose results to both players
    if(ch=='O' || ch=='X'){
    try{

    for(int i=0; i < numPlayers; i++){
    players[i].output.writeUTF("The Winner is: " +String.valueOf(winner) +"\n"
    +"The loser is: "+String.valueOf(loser));
    }
    }
    catch(IOException e){
    e.printStackTrace();
    }

    }//if
    }

    //tell players that game is over, so that new game button can be enabled
    //by players
    public void gameEnded()
    {
    try{
    for (int i=0; i < numPlayers; i++){
    players[i].output.writeUTF("Server: Game Ended");
    }
    }
    catch(IOException e){
    e.printStackTrace();
    }

    }

    private void startNewGame()
    {
    String str="";

    try{
    for (int i=0; i < numPlayers; i++){
    players[i].output.writeUTF("Server: New Game");
    }

    //If a game is played with server, X always moves first
    //If a game is not played with server,
    //if X wins or the game is a draw, O willl now make a move first
    //Otherwise, X will take a move first

    if(playWithServer){
    str="Grant Turn X";
    }
    else{
    if(winnerId=='X' || winnerId=='E'){
    str="Server: Grant Turn O";
    }
    else{
    str="Server: Grant Turn X";
    }

    }//end if else

    for (int i=0; i < numPlayers; i++){
    players[i].output.writeUTF(str);
    }
    }
    catch(IOException e){
    e.printStackTrace();
    }

    //Reset variables
    for(int i=0; i<9; i++){
    board[i]=' ';
    }
    //playWithServer=false;
    //serverTurn=false;
    draw=false;
    gameOver=false;
    winnerId='E';
    output.setText("");
    }

    //send message sent to both players when one of the player send a msg
    public void sendMsgToPlayers(String msg, int num)
    { String str="";
    if(num==SERVER){//server
    str="Server: ";
    }
    else{
    str=(num==0 ? "Player 1: " : "Player 2: ");
    }
    str=str+msg;

    try{
    for (int i=0; iplayers[i].output.writeUTF(str);
    }
    }
    catch(IOException e){}

    }

    // Recieve the message from client and interpret it
    public void checkMessage(String msg, int number)
    {
    String str="";
    if(msg.equals("Server: New Game")){
    System.out.println("Player "+ (number+1)+ " requested for a new game");
    startNewGame();
    }
    else if(msg.equals("Server: End Game")){
    str="Player "+ (number+1)+ " terminates the game. Game Ended\n";
    sendMsgToPlayers(str, SERVER);

    for (int i=0; iplayers[i].destroy();
    }
    }
    else{ //probably a message
    sendMsgToPlayers(msg, number);
    }
    }

    //This allow the server to decide whether to move randomly or selectively..
    public void serverMove()
    {
    int loc;
    loc=randomMove();
    output.append("Server loc: "+loc+ "\n");
    board[loc] ='O';
    players[currentPlayer].serverHasMoved(loc);
    players[currentPlayer].pause=false;
    }

    //This allows the computer to pick a board location randomly
    private int randomMove()
    {
    int num, moveLoc=-1;
    boolean loop=true;

    do{
    num=(int)(Math.random()*;

    if(board[num]==' '){
    loop=false;
    moveLoc=num;
    }

    }while(loop==true);

    return moveLoc;
    }

    // Player class to manage each Player as a thread
    class Player extends Thread {

    public Socket connection;
    public DataInputStream input;
    public DataOutputStream output;
    private TicTacToeServer control;
    private int number;
    private char mark;
    protected boolean threadSuspended = true;
    public boolean done=false;
    public boolean pause=false;

    public Player( Socket s, TicTacToeServer t, int num )
    {
    mark = ( num == 0 ? 'X' : 'O' );

    connection = s;

    try {
    input = new DataInputStream(
    connection.getInputStream() );
    output = new DataOutputStream(
    connection.getOutputStream() );
    }
    catch( IOException e ) {
    e.printStackTrace();
    System.exit( 1 );
    }

    control = t;
    number = num;
    }

    public void otherPlayerMoved( int loc )
    {
    try {
    output.writeUTF("Opponent moved");
    output.writeInt(loc);
    }
    catch ( IOException e ) { e.printStackTrace(); }
    }

    //this is used when there is only one player playing with the server
    public void currentPlayerMoved( int loc )
    {
    try {
    output.writeUTF("Current player moved");
    output.writeInt(loc);
    }
    catch ( IOException e ) { e.printStackTrace(); }
    }

    //this allow server to paint the board with '0' at desired location
    //also checks if game is over
    public void serverHasMoved(int loc)
    {
    try {
    output.writeUTF("Server moved");
    output.writeInt(loc);
    checkIfGameOver('O');
    }
    catch (IOException e) {}

    }

    //check if game is over
    public void checkIfGameOver(char mark)
    {
    if (control.gameOver(mark)){
    if(control.draw==true){
    mark='D';
    }
    control.gameOver=true;
    control.sendResults(mark);
    control.gameEnded();

    }//if

    }

    public void run()
    {
    int choice=-1;

    try {
    control.display( "Player " +
    (number == 0 ? 'X' : 'O' ) + " connected");

    output.writeChar(mark);
    output.writeUTF( "Player " +
    ( number == 0 ? "X connected\n" :
    "O connected, please wait\n" ) );

    // wait for another player to arrive
    if ( mark == 'X') {

    choice=JOptionPane.showConfirmDialog(null,
    "Play with the server as no player is connected at the moment?",
    "Play with Server?",JOptionPane.YES_NO_OPTION);

    if(choice==0){
    output.writeUTF("Server: Grant Turn X");
    control.playWithServer=true;
    control.numPlayers--;
    }
    else{
    output.writeUTF( "Waiting for another player" );
    try {
    synchronized( this ) {
    while ( threadSuspended )
    wait();
    }
    }
    catch ( InterruptedException e ) {
    e.printStackTrace();
    }
    output.writeUTF("Other player connected. Your move." );
    }//end if else
    }//if

    // Play game
    while (!done) {

    int loc=-1;

    if(!pause){ //if it is not player turn
    String msg = input.readUTF();
    //if parse fails then it is a message
    try{
    loc=Integer.parseInt(msg);
    }
    catch(NumberFormatException ex){
    control.checkMessage(msg, number);
    }
    }//if

    if(loc!=-1 && !pause){ //not a message

    if(control.validMove(loc, number)) {
    control.display("loc: " + loc);
    output.writeUTF("Valid move.");

    checkIfGameOver(mark);

    if(control.playWithServer){

    if(!control.gameOver){
    pause=true;
    control.serverMove();
    }
    pause=false;
    }//if
    }
    else{
    output.writeUTF( "Invalid move, try again" );
    }//end if else

    }//if

    }//while

    } //try
    catch( IOException e ) {
    e.printStackTrace();
    System.exit( 1 );
    }

    }//run

    }//class TicTacToeServer
