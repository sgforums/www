    $string = 'thisabcdunnodefmememe';
    $pattern     = '/^(.*)abc(.*)def(.*)$/i';
    $replacement = '$1cba$2fed$3';
    echo     preg_replace($pattern, $replacement, $string);

    Should give you 'thiscbadunnofedmememe'. I didn't test it but should be correct.
