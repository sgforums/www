    Basic command:

    Dim objConn As OleDbConnection = New OleDbConnection
    Dim objComm As OleDbCommand = New OleDbCommand
    Dim objDataSet As DataSet = New DataSet
    Dim strSQLCommand As String

    objConn.ConnectionString = "Provider=SQLOLEDB.1;Data Source=test;" & _
    "Initial Catalog=test;user id=user1;password=12345678;"

    strSQLCommand = "SELECT * FROM tblTest"

    objComm.CommandText = strSQLCommand
    objComm.Connection = objConn

    Dim objAdapter As OleDbDataAdapter = New OleDbDataAdapter
    objAdapter.SelectCommand = objComm
    objAdapter.SelectCommand.Connection.Open()

    objAdapter.Fill(objDataSet, "Data" )
    grdData.DataSource = objDataSet.Tables("Data" )
    grdData.DataBind()

    objAdapter.SelectCommand.Connection.Close()
