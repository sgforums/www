    Original Route = Pasir Ris Drive 12 -> TPE -> KPE -> Tampines Rd (after KPE)

    Diverted Route = Pasir Ris Drive 12 -> Tampines Ave 10 -> Tampines Rd (entering at the junction near the temples / where svc 72 turns)
