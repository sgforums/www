    for those who know IRC scripting, this is how it goes:

    alias liccheck {

    /.set %licchk_inputPrefix $1
    /.set %licchk_inputNumber $2

    /.set %licchk_SBS_CycledOrder ZXTRMKHECAYUSPLJGDB
    /.set %licchk_TIB_CycledOrder ECAYUSPLJGDBZXTRMKH

    if %licchk_inputNumber isnum 1-9999 {
    if ( ( %licchk_inputPrefix == SBS ) || ( %licchk_inputPrefix == TIB ) ) {
    if %licchk_inputNumber isnum 1000-9999 {
    /.set %licchk_inputNumber1 $mid(%licchk_inputNumber,1,1)
    /.set %licchk_inputNumber2 $mid(%licchk_inputNumber,2,1)
    /.set %licchk_inputNumber3 $mid(%licchk_inputNumber,3,1)
    /.set %licchk_inputNumber4 $mid(%licchk_inputNumber,4,1)
    }
    if %licchk_inputNumber isnum 100-999 {
    /.set %licchk_inputNumber1 0
    /.set %licchk_inputNumber2 $mid(%licchk_inputNumber,1,1)
    /.set %licchk_inputNumber3 $mid(%licchk_inputNumber,2,1)
    /.set %licchk_inputNumber4 $mid(%licchk_inputNumber,3,1)
    }
    if %licchk_inputNumber isnum 10-99 {
    /.set %licchk_inputNumber1 0
    /.set %licchk_inputNumber2 0
    /.set %licchk_inputNumber3 $mid(%licchk_inputNumber,1,1)
    /.set %licchk_inputNumber4 $mid(%licchk_inputNumber,2,1)
    }
    if %licchk_inputNumber isnum 1-9 {
    /.set %licchk_inputNumber1 0
    /.set %licchk_inputNumber2 0
    /.set %licchk_inputNumber3 0
    /.set %licchk_inputNumber4 $mid(%licchk_inputNumber,1,1)
    }

    /.set %licchk_inputNumberChecksum $calc( %licchk_inputNumber1 * 12 + %licchk_inputNumber2 * 2 + %licchk_inputNumber3 * 11 + %licchk_inputNumber4 * 1 )


    /.set %licchk_inputNumberChecksum $calc( %licchk_inputNumberChecksum % 19 )
    if ( %licchk_inputNumberChecksum == 0 ) {
    /.set %licchk_inputNumberChecksum 19
    }


    if ( %licchk_inputPrefix == SBS ) {
    /.set %licchk_outputSuffix $mid(%licchk_SBS_CycledOrder, %licchk_inputNumberChecksum, 1)
    }
    if ( %licchk_inputPrefix == TIB ) {
    /.set %licchk_outputSuffix $mid(%licchk_TIB_CycledOrder, %licchk_inputNumberChecksum, 1)
    }

    /.return %licchk_outputSuffix

    }
    else {
    /.return $null
    }
    }
    else {
    /.return $null
    }

    }

    EXPLANATION:
    for e.g. SBS9846, take thousands place times 12 (cos from SBS0 shift by 12 to the right to get suffix of SBS1000, and so on), take hundreds place times 2, take tens place times 11, take ones place times 1, add them up and find the remainder when divided by 19 (number of suffixes in cycled order).. find the corresponding position in the SBS's cycled ordering given above..

    1. mutliply each digit by their 'reference' number and add up:
    9 * 12 + 8 * 2 + 4 * 11 + 6 * 1 = 174

    2. divide by 19 and get remainder.
    if it is divisible by 19 exactly, take it as 19.
    174 divided by 19, the remainder is 3 ( 9 * 19 + 3)

    3. now, SBS's cycled order is "ZXTRMKHECAYUSPLJGDB" (SBS1 is Z, 2 is X etc.) and the 3th alphabet is T.. thus T is the suffix for SBS9846.

    A bug has been found, related to the fact that 0 remainder does not mean taking 0th alphabet but rather the last (19th) alphabet and an additional 3 lines of code are added.. the old bug meant numbers with suffix "B" had "Z" returned instead for SBS and suffix "H" had "E" returned instead for TIB.. users can download the new text file and overwrite the existing one with mIRC closed (important!)! then you can launch mIRC and use the new revised version.. download it at the same location here..
