    y = 2x + 1

    x^2 - 2x(2x+1) + (2x+1)^2 = 1

    x^2 - 4x^2 - 2x + 4x^2 + 4x + 1 = 1

    x^2 + 2x = 0

    x(x+2) = 0

    x = 0, y = 2(0) +1
    (x,y) = (0, 1)

    or x + 2 = 0
    x = -2 , y = 2(-2) +1
    (x,y) = (-2, -3)
