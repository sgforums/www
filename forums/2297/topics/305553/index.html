    (3x)lg 3 = (4x)lg 4

    Taking lg,
    lg 3 lg 3x = lg 4 lg 4x
    lg 3(lg 3 + lg x) = lg 4(lg 4+lg x)
    (lg 3)2 - (lg 4)2 = -lg x(lg 3 - lg 4)
    -lg x = lg 3 + lg 4 = lg 12
    lg x = lg 12-1 = lg (1/12)
    x = 1/12
