    y = (14 + 31.87)/82 or (14 - 31.87)/82
    y = 0.559 or - 0.218
    and so when y = 0.559
    2x + 2.795 = 6
    x = (6 - 2.795)/2 = 1.603
    or when y = - 0.218
    2x - 1.09 = 6
