    aim 0 // aimbot on at defualt or not
    aimthru 0 // if you wish to aimthru walls make this 1
    ahead 1 // prediction
    aspeed 1
    botfov 5 // the "range" around your crosshair to search for a target (0 - 360)
    box 0 // draw a box on enemys, 2 options
    distance 0 // draw distance on enemys
    duck 1 // makes you duck-jump always
    duck_f 4 // the ammount of "depth" you want on your aim
    duck_h 24 // the height you want the aimbot on ducking enemys
    duck_r 2.5 // the ammount of right you wish to use
    esp 1 // show esp
    flash 1 // dont show flashbangs
    glow 1 // makes people glow admin mod style
    jump 1 // makes you be able to "bunny hop"
    mode 1 // type of mode you wish to aim with , mode 0 = distance | mode 1 = fov
    name 1 // show names on players
    recoil 1 //a mount of recoil compensation to use
    rspeed 1 // speed bound on the shift key
    shoot 0 // autoshoot
    radar 1 // show radar
    smoke 0 // hides smoke if enabled
    soundesp 1 // shows ESP thats based on sound
    stand_f 5 // the ammount "depth you want on your aim
    stand_h 21 // the height you want to aim at an enemy
    stand_r 1.5 // the ammount of right you wish to use
    t_glow 1 // make your current targets color change to white
    team 0 // what team you wish to aim at, leave this 0
    weapon 1 // show weapon names on people
    thirdperson 0 // note esp is messed
    fullbright 0 // this works now, no crashies
    bind 1 // binding style
    //nopatch 1 // uncomment this if detected
    radar_centerx 105
    radar_centery 320
    radar_size 60 // size of radar
    radar_range 1500 // no need to touch really
    switch 1 // this is for auto pistol switch, enabled by defualt.
    infohud 0 // shows stats on your hud, disabled by defualt
    pistol 1 // autofire with pistol (you can hold mouse1 down and it keeps shooting) enabled by defualt.
    crosshair 1 // crosshair obviously...
