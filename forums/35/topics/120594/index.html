    Try this :

    <%

    Set objNewMail = Server.CreateObject( "CDONTS.NewMail " )

    objNewMail.To = "info@sgcontrol.com.sg"
    objNewMail.From = Request.form( "email" )

    ' message subject
    objNewMail.Subject = "The following message has been entered from ABC website."

    ' message body
    objNewMail.Body = "Message Detail:" & chr(10) &_
    "Name: " & Request.form( "name" ) & chr(10) &_
    "Company: " & Request.form( "company" ) & chr(10) &_
    "Email: " & Request.form( "email" ) & chr(10) &_
    "Phone: " & Request.form( "phone" ) & chr(10) &_
    "Nature of feedback: " & Request.form( "nature" ) & chr(10) &_
    "Message: " & Request.form( "message" )

    strErr = ""
    bSuccess = False
    On Error Resume Next ' catch errors
    objNewMail.Bcc = ""
    objNewMail.Send ' send message
    Set objNewMail = Nothing

    If Err <> 0 Then ' error occurred
    strErr = Err.Description
    else
    bSuccess = True
    End If

    ' if you have any thank you message for the user
    Response.Redirect "thankyou.htm"

    %>
